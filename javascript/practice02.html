map - 新しく配列を作る

mapの使い方

returnで新しい配列にして返してくれる。

      var numbers = [1,2,3];

      var doubles = numbers . map ( function ( number ){
        return number * 2;
      });

      ーーーーー
      [ 2 , 4 , 6 ]

関数を作って使ってみる

関数のpropertyに指定した、colorだけが新たな配列に入って返ってくる。

      var paints = [{ color: '赤', fruit: 'apple' }, { color: 'オレンジ', fruit: 'orange' }, { color: '黄', fruit: 'remon' }];

      function pluck ( array , property ){
        return array . map ( function ( arr ) {
          return arr [ property ]
        });
      }

      pluck ( paints , ' color ' );

      ーーーーー
      [ '赤' , 'オレンジ' , '黄 ]
