filter - フィルタリング

filterの使い方

returnで、一致するものを返してくれる。

      var products = [
        { name: 'きゅうり', type: '野菜' , price: 14},
        { name: 'バナナ', type: 'フルーツ', price: 6 },
        { name: 'セロリ', type: '野菜', price: 20 },
        { name: 'オレンジ', type: 'フルーツ', price: 25 }
      ];

      products . filter ( function ( product ) {
        return product . type === 'フルーツ'  &&  product . price < 10;
      });

      ーーーーー
      { name : 'バナナ', type : 'フルーツ' }

関数を作って使ってみる

配列の中から一致するものだけを返してくれる。

      var post = { id: 4 , title: '初めての投稿' };
      var comments = [
        { postId: 4 , content: 'コメント１' },
        { postId: 3 , content: 'コメント２' },
        { postId: 4 , content: 'コメント３' }
      ];

      function commentsForPost ( post , comments ) {
        return comments . filter ( function ( comment ) {
          return comment . postId === post . id;
        });
      }

      commentsForPost( post , comments )
      ーーーーー
      [{ postId: 4 , content: 'コメント１' },{ postId: 4 , content: 'コメント３' }]
