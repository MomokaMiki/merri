every/some - 論理積/論理和

everyの使い方

全てが条件にあっているかどうかを判定する

      var computers = [
        { name : 'Apple', ram : 24 },
        { name : 'Compaq', ram : 4 },
        { name : 'Acer', ram : 16 }
      ]

      computers . every ( function ( computer ) {
        return computer . ram >= 16;
      })

      ーーーーー
      false

someの使い方

ひとつでも条件にあっているものがあるかどうか

      var computers = [
        { name : 'Apple', ram : 24 },
        { name : 'Compaq', ram : 4 },
        { name : 'Acer', ram : 16 }
      ]
      
      computers . some ( function ( computer ) {
        return computer . ram >= 16;
      })
      
      ーーーーー
      true
